    result =`Two real root: ${root1.toFixed(2)} and ${root2.toFixed(2)}`;
    }
    else if (discriminant == 0){
        let root = -b / (2*a );
        result = `one real root : ${root.toFixed(2)}`;
    }

        result = "no real root";
    }
